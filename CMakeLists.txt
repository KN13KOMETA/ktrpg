cmake_minimum_required(VERSION 3.20 FATAL_ERROR)

# TODO: read about cmake presets

cmake_path(SET SOURCE_FILES_DIR "${CMAKE_SOURCE_DIR}/src")

include("cmake/functions.cmake")

# Always export compile commands (needs for clangd)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

project(
  c_text_rpg
  VERSION 3.0.0
  DESCRIPTION "Text RPG written in C"
  HOMEPAGE_URL https://github.com/KN13KOMETA/c_text_rpg
  LANGUAGES C
)

include("cmake/version.cmake")

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE "Debug")
endif()

# Force CMake to use c99 instead of gnu99
set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

# Embed lua scripts to source files
generate_header_from_lua(${SOURCE_FILES_DIR}/lua/init.lua)

add_executable(
  ${PROJECT_NAME}
  ${GENERATED_C_FILES}
  src/functions.c
  src/project_options.c
  src/tgg/components.c
  src/tgg/systems.c
  src/tgg/tgg_lua_module.c
  src/tgg/tgg.c
  src/game/ecs/components.c
  src/game/ecs/systems.c
  src/main.c
)

# Autogenerated files
add_custom_target(
  generated_files
  ALL
  DEPENDS ${GENERATED_HEADERS} ${GENERATED_C_FILES}
)
add_dependencies(${PROJECT_NAME} generated_files)

include("cmake/libraries.cmake")

set_target_properties(
  ${PROJECT_NAME}
  PROPERTIES
    OUTPUT_NAME
      "${PROJECT_NAME}_${PROJECT_FULL_VERSION}$<$<CONFIG:DEBUG>:_debug>"
)

# Turning on a lot compiler checks
# "В семье не без урода" MSVC flags differs from Clang/GCC
if(CMAKE_C_COMPILER_ID MATCHES "MSVC")
  target_compile_options(
    ${PROJECT_NAME}
    PRIVATE /W4 /permissive- $<$<CONFIG:RELEASE>:/WX>
  )
else()
  target_compile_options(
    ${PROJECT_NAME}
    PRIVATE
      -xc
      -Wall
      -Wextra
      -Wpedantic
      -Wconversion
      -Wmissing-prototypes
      -Wmissing-variable-declarations
      $<$<CONFIG:RELEASE>:-Werror>
  )
endif()

option(ENABLE_WEVERYTHING "Enable Clang -Weverything" OFF)

if(ENABLE_WEVERYTHING)
  if(CMAKE_C_COMPILER_ID MATCHES "Clang")
    target_compile_options(
      ${PROJECT_NAME}
      PRIVATE
        -Weverything
        # Used for c90 is lower
        -Wno-declaration-after-statement
        # Used for C++
        -Wno-unsafe-buffer-usage
    )
  endif()
endif()

target_compile_definitions(
  ${PROJECT_NAME}
  PUBLIC
    PROJECT_NAME=\"${PROJECT_NAME}\"
    PROJECT_FULL_VERSION=\"${PROJECT_FULL_VERSION}\"
    PROJECT_DESCRIPTION=\"${PROJECT_DESCRIPTION}\"
    PROJECT_HOMEPAGE_URL=\"${PROJECT_HOMEPAGE_URL}\"
    $<$<CONFIG:DEBUG>:DEBUG>
)

get_property(
  TARGET_COMPILE_OPTS
  TARGET ${PROJECT_NAME}
  PROPERTY COMPILE_OPTIONS
)
message(STATUS "Compile options for ${PROJECT_NAME}: ${TARGET_COMPILE_OPTS}")

include("cmake/target/todo.cmake")
